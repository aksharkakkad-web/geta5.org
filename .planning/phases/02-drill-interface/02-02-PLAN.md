    **Fully replace** the stub at `app/[subject]/drills/page.tsx` with:

    ```typescript
    'use client'

    import { useState } from 'react'
    import { use } from 'react'
    import UnitSelector from '@/components/drill/UnitSelector'
    import DrillSession from '@/components/drill/DrillSession'
    import DrillResults from '@/components/drill/DrillResults'
    import type { DrillView, SessionState } from '@/utils/drillSession'

    interface DrillsPageProps {
      params: Promise<{ subject: string }>
    }

    export default function DrillsPage({ params }: DrillsPageProps) {
      const { subject } = use(params)
      const [view, setView] = useState<DrillView>('unit-select')
      const [session, setSession] = useState<SessionState | null>(null)

      const handleStart = (newSession: SessionState) => {
        setSession(newSession)
        setView('session')
      }

      const handleComplete = (finalSession: SessionState) => {
        setSession(finalSession)
        setView('results')
      }

      const handleRetry = (retrySession: SessionState) => {
        setSession(retrySession)
        setView('session')
      }

      const handleUnitSelect = () => {
        setSession(null)
        setView('unit-select')
      }

      return (
        <main style={{ padding: '24px', maxWidth: '960px', margin: '0 auto' }}>
          {view === 'unit-select' && (
            <UnitSelector subject={subject} onStart={handleStart} />
          )}
          {view === 'session' && session && (
            <DrillSession session={session} subject={subject} onComplete={handleComplete} />
          )}
          {view === 'results' && session && (
            <DrillResults
              session={session}
              subject={subject}
              onRetry={handleRetry}
              onUnitSelect={handleUnitSelect}
            />
          )}
        </main>
      )
    }
    ```

    **CRITICAL: params handling for Next.js 15+.**
    In Next.js 15, even `'use client'` page components receive `params` as a `Promise`. Use React's `use()` hook to unwrap it synchronously inside the client component. This is the correct pattern — do NOT use `await` (which requires async function) and do NOT type params as a plain object (which causes the Next.js 15 warning). The `use()` hook from React 19 handles Promise unwrapping in client components.

    **Key behaviors:**
    - URL never changes during session — all view transitions are React state changes
    - Browser back exits to subject hub (default Next.js behavior — no custom history management needed)
    - `handleUnitSelect` resets session to null so UnitSelector re-fetches fresh
    - The `<main>` wrapper provides consistent padding and max-width for all views
